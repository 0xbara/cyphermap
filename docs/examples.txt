WHERE n.name = 'Alice' AND head(db.labels(n)) = 'test'

WHERE n.name = 'Alice' AND left(head(db.labels(n)), 1) = 't'
RETURN m.name

MATCH (n:Person)-[:FRIENDS_WITH]->(m:Person)
WHERE n.name = 'Alice' AND substring(head(db.labels(n)), 1, 2) = 'te'
RETURN m.name

RETURN m.name
